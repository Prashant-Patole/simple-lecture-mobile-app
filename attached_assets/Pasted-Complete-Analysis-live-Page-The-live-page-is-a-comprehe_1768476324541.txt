Complete Analysis: /live Page
The /live page is a comprehensive Class Schedule Management System that displays live classes, upcoming classes, attendance tracking, and class recordings for enrolled students.

1. DATABASE SCHEMA
The page uses 7 main tables:

scheduled_classes - One-time scheduled classes
Column	Type	Description
id	uuid	Primary key
course_id	uuid	Foreign key to courses
teacher_id	uuid	Foreign key to teacher_profiles
subject	text	Subject name
subject_id	uuid	Foreign key to popular_subjects
chapter_id	uuid	Foreign key to subject_chapters
topic_id	uuid	Foreign key to subject_topics
scheduled_at	timestamptz	When class is scheduled
duration_minutes	integer	Class duration
meeting_link	text	YouTube/BBB meeting link
room_number	text	Physical room location
is_live	boolean	Whether class is currently live
is_cancelled	boolean	Whether class is cancelled
live_started_at	timestamptz	When live started
live_ended_at	timestamptz	When live ended
bbb_meeting_id	text	BigBlueButton meeting ID
course_timetables - Recurring weekly classes
Column	Type	Description
id	uuid	Primary key
course_id	uuid	Foreign key to courses
subject_id	uuid	Foreign key to popular_subjects
instructor_id	uuid	Foreign key to teacher_profiles
day_of_week	integer	0-6 (Sunday-Saturday)
start_time	time	Start time (HH:MM:SS)
end_time	time	End time (HH:MM:SS)
meeting_link	text	YouTube/BBB meeting link
room_number	text	Physical room location
is_active	boolean	Whether entry is active
class_attendance - Student attendance records
Column	Type	Description
id	uuid	Primary key
scheduled_class_id	uuid	Foreign key to scheduled_classes
student_id	uuid	Student user ID
status	text	'present', 'absent', 'late'
joined_at	timestamptz	When student joined
left_at	timestamptz	When student left
duration_seconds	integer	Time spent in class
class_recordings - Processed class recordings
Column	Type	Description
id	uuid	Primary key
scheduled_class_id	uuid	Link to class
duration_seconds	integer	Recording length
processing_status	text	'pending', 'processing', 'ready', 'failed'
available_qualities	jsonb	['360p', '480p', '720p', '1080p']
b2_hls_720p_path	text	HLS video path
bunny_video_guid	text	Bunny CDN video ID
cdn_base_url	text	CDN URL for streaming
video_watch_progress - User video progress tracking
Column	Type	Description
id	uuid	Primary key
recording_id	uuid	Foreign key to class_recordings
user_id	uuid	User ID
progress_seconds	integer	Seconds watched
progress_percent	numeric	Percentage completed
completed	boolean	Whether fully watched
enrollments - Student course enrollments
Column	Type	Description
student_id	uuid	User ID
course_id	uuid	Course ID
is_active	boolean	Active enrollment
2. API ENDPOINTS
Supabase REST API Calls

// 1. GET STUDENT ENROLLMENTS
GET /rest/v1/enrollments
  ?student_id=eq.{userId}
  &is_active=eq.true
  &select=course_id,batch_id,enrolled_at

// 2. GET SCHEDULED CLASSES (one-time classes)
GET /rest/v1/scheduled_classes
  ?course_id=in.({courseIds})
  &is_cancelled=eq.false
  &scheduled_at=gte.{todayStart}
  &scheduled_at=lte.{weekEnd}
  &order=scheduled_at.asc
  &select=id,subject,scheduled_at,duration_minutes,meeting_link,is_live,
          course_id,subject_id,chapter_id,topic_id,
          courses(id,name),
          popular_subjects(id,name),
          subject_chapters(id,title),
          subject_topics(id,title),
          teacher_profiles(id,full_name,avatar_url)

// 3. GET COURSE TIMETABLES (recurring classes)
GET /rest/v1/course_timetables
  ?course_id=in.({courseIds})
  &is_active=eq.true
  &select=id,day_of_week,start_time,end_time,room_number,meeting_link,
          course_id,subject_id,
          courses(id,name),
          popular_subjects(id,name),
          teacher_profiles(id,full_name,avatar_url)

// 4. GET FILTER OPTIONS - Courses
GET /rest/v1/courses
  ?id=in.({courseIds})
  &order=name
  &select=id,name

// 5. GET FILTER OPTIONS - Subjects
GET /rest/v1/course_subjects
  ?course_id=in.({courseIds})
  &select=course_id,subject_id,
          popular_subjects(id,name)

// 6. GET FILTER OPTIONS - Chapters
GET /rest/v1/subject_chapters
  ?subject_id=in.({subjectIds})
  &order=sequence_order
  &select=id,title,subject_id

// 7. GET ATTENDANCE RECORDS
GET /rest/v1/class_attendance
  ?student_id=eq.{userId}
  &select=id,status,duration_seconds,
          scheduled_class:scheduled_classes(id,course_id,scheduled_at,
            courses(id,name))

// 8. GET CLASS RECORDINGS
GET /rest/v1/class_recordings
  ?processing_status=eq.ready
  &order=created_at.desc
  &select=*,
          scheduled_class:scheduled_classes(id,subject,scheduled_at,course_id,
            course:courses(name))

// 9. GET VIDEO WATCH PROGRESS
GET /rest/v1/video_watch_progress
  ?recording_id=eq.{recordingId}
  &select=*

// 10. UPDATE VIDEO WATCH PROGRESS
POST /rest/v1/video_watch_progress
  UPSERT { recording_id, user_id, progress_seconds, progress_percent, completed }

6. KEY FEATURES & IMPLEMENTATIONS
Real-time Updates

// Live class updates subscription
useEffect(() => {
  const channel = supabase
    .channel('live-class-updates')
    .on(
      'postgres_changes',
      {
        event: '*',
        schema: 'public',
        table: 'scheduled_classes',
      },
      () => {
        refetch();
        queryClient.invalidateQueries({ queryKey: ['unified-live-classes'] });
      }
    )
    .subscribe();

  return () => {
    supabase.removeChannel(channel);
  };
}, [refetch, queryClient]);
Unified Class Processing
Combines two data sources:

scheduled_classes - One-time scheduled classes with exact datetime
course_timetables - Recurring weekly entries converted to actual dates

// Timetable to actual date conversion
function getOccurrences(entry: any, daysAhead: number = 7): Date[] {
  const today = new Date();
  const occurrences: Date[] = [];
  
  for (let i = 0; i < daysAhead; i++) {
    const date = addDays(today, i);
    if (date.getDay() === entry.day_of_week) {
      occurrences.push(date);
    }
  }
  return occurrences;
}

// Class is considered live if:

3.2 Attendance API (useAttendance):


-- Step 1: Get scheduled classes for enrolled courses (last 30 days)
SELECT id, scheduled_at FROM scheduled_classes
WHERE course_id IN (enrolled_course_ids)
  AND scheduled_at >= {30_days_ago}
  AND scheduled_at <= NOW()
  AND is_cancelled = false;

-- Step 2: Get student's attendance records
SELECT *, scheduled_class:scheduled_classes(scheduled_at, subject)
FROM class_attendance
WHERE student_id = {user_id}
  AND scheduled_class_id IN (class_ids);
Calculated Values:

percentage: (present_count / total_classes) * 100
present: Count of status = 'present'
total: Total scheduled classes
3.3 DPT API (useDPT):


SELECT * FROM dpp_topic_submissions
WHERE student_id = {user_id}
ORDER BY test_date DESC;
Response:

Calculated Values:

streak: Count of consecutive days with submissions starting from today
averageScore: sum(score/total_questions) / count * 100

weeklyData: Array of last 7 days with completion status

4. EnrolledCoursesSection Component
File: src/components/dashboard/EnrolledCoursesSection.tsx

Purpose: Displays enrolled courses with subjects in a tabbed interface.

Hook Used: useDashboardCourseDetails()

API Calls (6 batch queries):


-- Batch 1: Get enrolled course IDs
SELECT course_id, batch_id, enrolled_at FROM enrollments
WHERE student_id = {user_id} AND is_active = true;

-- Batch 2: Get course details
SELECT id, name, slug, thumbnail_url FROM courses
WHERE id IN (enrolled_course_ids);

-- Batch 3: Get course subjects
SELECT course_id, subject_id, display_order,
       popular_subjects(id, name, description, thumbnail_url)
FROM course_subjects
WHERE course_id IN (enrolled_course_ids)
ORDER BY display_order;

-- Batch 4: Get all chapters for subjects
SELECT id, subject_id FROM subject_chapters
WHERE subject_id IN (all_subject_ids);

-- Batch 5: Get student progress
SELECT chapter_id FROM student_progress
WHERE student_id = {user_id}
  AND is_completed = true
  AND chapter_id IN (all_chapter_ids);

-- Batch 6: Get assignments and submissions
SELECT id, chapter_id, course_id FROM assignments
WHERE course_id IN (enrolled_course_ids) AND is_active = true;

SELECT assignment_id FROM assignment_submissions
WHERE student_id = {user_id} AND assignment_id IN (assignment_ids);

5. UpcomingClasses Component
File: src/components/dashboard/UpcomingClasses.tsx

Purpose: Displays next 5 scheduled classes with join functionality.

Hook Used: useUpcomingClasses()

API Calls:


-- Step 1: Get enrolled course IDs
SELECT course_id FROM enrollments
WHERE student_id = {user_id} AND is_active = true;

-- Step 2: Get scheduled classes
SELECT id, subject, subject_id, notes, scheduled_at, 
       duration_minutes, meeting_link, is_live, teacher_id, course_id
FROM scheduled_classes
WHERE course_id IN (enrolled_course_ids)
  AND scheduled_at >= NOW()
  AND is_cancelled = false
ORDER BY scheduled_at ASC
LIMIT 10;

-- Step 3: Batch fetch related data
SELECT id, name FROM popular_subjects WHERE id IN (subject_ids);
SELECT id, full_name, avatar_url FROM teacher_profiles WHERE id IN (teacher_ids);
SELECT id, name FROM courses WHERE id IN (course_ids);

7. NotificationModal Component
File: src/components/dashboard/NotificationModal.tsx

Purpose: Shows all notices/notifications in a modal dialog.

Hook Used: useNotices()

API Calls:


-- Get notices
SELECT * FROM notices
WHERE is_active = true
ORDER BY created_at DESC;

-- Get read status
SELECT notice_id FROM notice_reads
WHERE user_id = {user_id};

-- Mark as read (UPSERT)
INSERT INTO notice_reads (notice_id, user_id)
VALUES ({notice_id}, {user_id})
ON CONFLICT DO UPDATE SET notice_id = EXCLUDED.notice_id;

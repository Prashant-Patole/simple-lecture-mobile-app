interface StudentAssignment {
  id: string;
  title: string;
  description: string | null;
  due_date: string | null;
  total_marks: number;
  passing_marks: number;
  duration_minutes: number | null;
  chapter_id: string | null;
  topic_id: string | null;
  status: 'pending' | 'submitted' | 'graded';
  score: number | null;
  percentage: number | null;
  submitted_at: string | null;
  graded_at: string | null;
}

async function getStudentAssignments(
  supabase: SupabaseClient,
  subjectId: string,
  userId: string,
  chapterId?: string,
  topicId?: string
): Promise<StudentAssignment[]> {
  // Step 1: Fetch assignments
  let query = supabase
    .from('assignments')
    .select('id, title, description, due_date, total_marks, passing_marks, duration_minutes, chapter_id, topic_id')
    .eq('subject_id', subjectId)
    .eq('is_active', true)
    .order('created_at', { ascending: false });

  if (chapterId) query = query.eq('chapter_id', chapterId);
  if (topicId) query = query.eq('topic_id', topicId);

  const { data: assignments, error } = await query;
  if (error || !assignments) return [];

  // Step 2: Fetch submissions for these assignments
  const assignmentIds = assignments.map(a => a.id);
  const { data: submissions } = await supabase
    .from('assignment_submissions')
    .select('assignment_id, score, percentage, submitted_at, graded_at')
    .eq('student_id', userId)
    .in('assignment_id', assignmentIds);

  // Step 3: Merge data
  const submissionMap = new Map(
    (submissions || []).map(s => [s.assignment_id, s])
  );

  return assignments.map(assignment => {
    const submission = submissionMap.get(assignment.id);
    let status: 'pending' | 'submitted' | 'graded' = 'pending';
    if (submission) {
      status = submission.graded_at ? 'graded' : 'submitted';
    }

    return {
      ...assignment,
      status,
      score: submission?.score ?? null,
      percentage: submission?.percentage ?? null,
      submitted_at: submission?.submitted_at ?? null,
      graded_at: submission?.graded_at ?? null,
    };
  });
}

interface AssignmentQuestion {
  id: string;
  question: string;
  type: 'mcq' | 'true_false' | 'short_answer' | 'long_answer' | 'application';
  options?: string[];
  correct_answer?: string;
  marks: number;
  explanation?: string;
  image_url?: string;
}

async function getAssignmentQuestions(
  supabase: SupabaseClient,
  assignmentId: string
) {
  const { data, error } = await supabase
    .from('assignments')
    .select('*')
    .eq('id', assignmentId)
    .single();

  if (error || !data) return null;

  // Parse JSONB questions
  const rawQuestions = data.questions as any[];
  const questions: AssignmentQuestion[] = rawQuestions.map((q, index) => ({
    id: q.id || `q_${index}`,
    question: q.question || q.text || '',
    type: q.type || 'short_answer',
    options: q.options || [],
    correct_answer: q.correct_answer || q.answer || '',
    marks: q.marks || 1,
    explanation: q.explanation || '',
    image_url: q.image_url || null,
  }));

  return {
    id: data.id,
    title: data.title,
    description: data.description,
    instructions: data.instructions,

    passing_marks: data.passing_marks || 0,

    due_date: data.due_date,

4.3 Submit Assignment with AI Grading

async function submitAssignment(
  supabase: SupabaseClient,
  assignmentId: string,
  userId: string,
  answers: Record<string, { text?: string; imageUrl?: string }>,
  questions: AssignmentQuestion[],
  totalMaxMarks: number,
  startTime: number
) {
  // Separate MCQ and subjective questions
  const mcqTypes = ['mcq', 'true_false'];
  const mcqQuestions = questions.filter(q => mcqTypes.includes(q.type));
  const subjectiveQuestions = questions.filter(q => !mcqTypes.includes(q.type));

  // Grade MCQs locally
  const gradedAnswers: Record<string, any> = {};
  let totalMarks = 0;

  for (const q of mcqQuestions) {
    const studentAnswer = answers[q.id]?.text || '';
    const isCorrect = studentAnswer.trim().toLowerCase() === (q.correct_answer || '').trim().toLowerCase();
    const marksAwarded = isCorrect ? q.marks : 0;
    totalMarks += marksAwarded;

    gradedAnswers[q.id] = {
      question_id: q.id,
      text_answer: studentAnswer,
      image_url: answers[q.id]?.imageUrl,
      marks_awarded: marksAwarded,
      is_correct: isCorrect,
      feedback: isCorrect ? 'Correct!' : `Incorrect. Correct answer: ${q.correct_answer}`,
    };
  }

  // Grade subjective questions with AI
  if (subjectiveQuestions.length > 0) {
    const questionsToGrade = subjectiveQuestions.map(q => ({
      question_id: q.id,
      question: q.question,
      type: q.type,
      correct_answer: q.correct_answer || '',
      student_answer: answers[q.id]?.text || '',
      marks: q.marks,
    }));

    const { data: gradeResult, error } = await supabase.functions.invoke(
      'ai-grade-assignment',
      { body: { questions: questionsToGrade } }
    );

    if (!error && gradeResult?.grades) {
      for (const grade of gradeResult.grades) {
        const q = subjectiveQuestions.find(sq => sq.id === grade.question_id);
        if (q) {
          totalMarks += grade.marks_awarded;
          gradedAnswers[q.id] = {
            question_id: q.id,
            text_answer: answers[q.id]?.text || '',
            image_url: answers[q.id]?.imageUrl,
            marks_awarded: grade.marks_awarded,
            is_correct: grade.is_correct,
            feedback: grade.feedback,
          };
        }
      }
    }
  }

  // Calculate percentage and save
  const percentage = totalMaxMarks > 0 ? Math.round((totalMarks / totalMaxMarks) * 100) : 0;
  const timeTaken = Math.floor((Date.now() - startTime) / 1000);

  // Check for existing submission
  const { data: existing } = await supabase
    .from('assignment_submissions')
    .select('id')
    .eq('assignment_id', assignmentId)
    .eq('student_id', userId)
    .maybeSingle();

  const submissionData = {
    assignment_id: assignmentId,
    student_id: userId,
    answers: gradedAnswers,
    score: totalMarks,
    percentage,
    feedback: percentage >= 60 ? 'Good job!' : 'Keep practicing!',
    submitted_at: new Date().toISOString(),
    graded_at: new Date().toISOString(),
    time_taken_seconds: timeTaken,
  };

  if (existing) {
    await supabase.from('assignment_submissions').update(submissionData).eq('id', existing.id);
  } else {
    await supabase.from('assignment_submissions').insert(submissionData);
  }

  return { gradedAnswers, totalMarks, percentage };
}

function arrayBufferToBase64(buffer: ArrayBuffer): string {
  const bytes = new Uint8Array(buffer);
  let binary = '';
  const chunkSize = 8192;
  for (let i = 0; i < bytes.length; i += chunkSize) {
    const chunk = bytes.slice(i, i + chunkSize);
    binary += String.fromCharCode.apply(null, [...chunk]);
  }
  return btoa(binary);
}

  if (topicsError) throw topicsError;

  // Group topics by chapter
  return chapters.map(chapter => ({
    ...chapter,
    topics: topics.filter(t => t.chapter_id === chapter.id)
  }));
}

// Step 2: Get videos for TOPIC level
async function getTopicVideos(topic: Topic): Promise<VideoList> {
  const videos = [];

  // Source 1: Direct video from topic
  if (topic.video_id && topic.video_platform) {
    videos.push({
      id: `topic-direct-${topic.id}`,
      video_name: topic.title,
      language: 'english',
      video_platform: topic.video_platform,
      video_id: topic.video_id,
    });
  }

  // Source 2: Multi-language videos from topic_videos table
  const { data: additionalVideos } = await supabase
    .from('topic_videos')
    .select('*')
    .eq('topic_id', topic.id)
    .eq('is_active', true)
    .order('language')
    .order('display_order');

  if (additionalVideos) {
    videos.push(...additionalVideos);
  }

  return {
    videos,
    aiGeneratedVideoUrl: topic.ai_generated_video_url
  };
}

// Step 3: Get videos for CHAPTER level
function getChapterVideos(chapter: Chapter): VideoList {
  // Chapter level: Only uses chapter's own video and AI video
  // NO additional multi-language videos from topic_videos table
  
  const videos = [];
  
  if (chapter.video_id && chapter.video_platform) {
    videos.push({
      id: `chapter-direct-${chapter.id}`,
      video_name: `Ch ${chapter.chapter_number}: ${chapter.title}`,
      language: 'english',
      video_platform: chapter.video_platform,
      video_id: chapter.video_id,
    });
  }

  return {
    videos,
    aiGeneratedVideoUrl: chapter.ai_generated_video_url
  };
}

// Step 4: Build video embed URL
function getVideoEmbedUrl(video: Video): string {
  if (video.video_platform === 'youtube') {
    return `https://www.youtube.com/embed/${video.video_id}?autoplay=1`;
  } else if (video.video_platform === 'vimeo') {
    return `https://player.vimeo.com/video/${video.video_id}?autoplay=1`;
  }
  return '';
}

// Step 5: Get video thumbnail
function getVideoThumbnail(video: Video): string {
  if (video.video_platform === 'youtube') {
    return `https://img.youtube.com/vi/${video.video_id}/hqdefault.jpg`;
  }
  return ''; // Vimeo needs API call for thumbnail
}

3. API Queries Used
Query 1: Fetch Course with Related Data

const { data: course, isLoading } = useQuery({
  queryKey: ["program-detail", slug],
  queryFn: async () => {
    const { data, error } = await supabase
      .from("courses")
      .select(`
        *,
        course_categories (
          categories (
            id,
            name,
            slug
          )
        ),
        course_subjects (
          display_order,
          popular_subjects (
            id,
            name,
            slug,
            description,
            thumbnail_url
          )
        ),
        course_faqs (
          id,
          question,
          answer,
          display_order
        )
      `)
      .eq("slug", slug)        // Filter by URL slug
      .eq("is_active", true)   // Only active courses
      .single();               // Return single object

    if (error) throw error;
    return data;
  },
  enabled: !!slug,
});
What this fetches:

Data	Source Table	Join
Course details	courses	Direct
Categories	categories	Through course_categories
Subjects	popular_subjects	Through course_subjects
FAQs	course_faqs	Direct (course_id)
Query 2: Check User Enrollment

const { data: enrollment } = useQuery({
  queryKey: ["enrollment-check", course?.id],
  queryFn: async () => {
    // Get current logged-in user
    const { data: { user } } = await supabase.auth.getUser();
    if (!user || !course?.id) return null;

    // Check if user is enrolled in this course
    const { data } = await supabase
      .from("enrollments")
      .select("id, course_id")
      .eq("student_id", user.id)
      .eq("course_id", course.id)
      .eq("is_active", true)
      .maybeSingle();

    return data;
  },
  enabled: !!course?.id,
});

const isEnrolled = !!enrollment;

5. Database Schema Used
courses table

id, name, slug, description, short_description, detailed_description,
price_inr, original_price_inr, duration_months, thumbnail_url,
rating, review_count, student_count, is_active,
what_you_learn (JSONB), course_includes (JSONB)

id, course_id, category_id, created_at
course_subjects (Junction Table)

id, course_id, subject_id, display_order, created_at
course_faqs table

id, course_id, question, answer, display_order
enrollments table

id, student_id, course_id, is_active, created_at

MCQ Tab - Complete Technical Documentation
Overview
The MCQ (Multiple Choice Questions) tab in SimpleLecture works at two distinct levels:

Topic Level - Questions specific to a single topic within a chapter
Chapter Level - Questions that cover the entire chapter (not tied to any specific topic)
Database Schema

-- questions table structure (key fields)
CREATE TABLE questions (
  id UUID PRIMARY KEY,
  question_text TEXT NOT NULL,
  options JSONB,                    -- {"A": {"text": "..."}, "B": {"text": "..."}, ...}
  correct_answer TEXT NOT NULL,     -- "A", "B", "C", or "D"
  explanation TEXT,
  difficulty TEXT,                  -- "Low", "Medium", "Intermediate", "Advanced"
  marks INTEGER DEFAULT 1,
  question_format TEXT,             -- "single_choice", "multiple_choice", or NULL
  question_image_url TEXT,
  is_important BOOLEAN DEFAULT false,
  topic_id UUID REFERENCES topics(id),      -- NULL for chapter-level
  chapter_id UUID REFERENCES chapters(id),  -- Required for chapter-level
  created_at TIMESTAMPTZ
);
Key Difference: Topic vs Chapter Level
Aspect	Topic Level	Chapter Level
topic_id	Has a value (UUID)	NULL
chapter_id	May or may not have value	Required (UUID)
Scope	Specific to one topic	Covers entire chapter
Use Case	Practice after studying a topic	Chapter revision/assessment
Current Database Stats:

Chapter Level Questions: 171 (where topic_id IS NULL)
Topic Level Questions: 577 (where topic_id IS NOT NULL)

    .select("*")

// Get counts by difficulty for setup screen
const getQuestionCounts = async (topicId, chapterId, chapterOnly) => {
  let query = supabase
    .from("questions")
    .select("difficulty")
    .or("question_format.eq.single_choice,question_format.eq.multiple_choice,question_format.is.null")
    .not("options", "is", null);

  if (topicId) {
    query = query.eq("topic_id", topicId);
  } else if (chapterId && chapterOnly) {
    query = query.eq("chapter_id", chapterId).is("topic_id", null);
  } else if (chapterId) {
    query = query.eq("chapter_id", chapterId);
  }

  const { data } = await query;
  
  // Return counts object
  const counts = { all: 0, Low: 0, Medium: 0, Intermediate: 0, Advanced: 0 };
  data?.forEach(q => {
    counts.all++;
    if (counts[q.difficulty] !== undefined) counts[q.difficulty]++;
  });
  
  return counts;
};

// Complete MCQ fetching for mobile
class MCQService {
  constructor(supabaseClient) {
    this.supabase = supabaseClient;
  }

  // Transform options from JSONB to array
  transformOptions(dbOptions) {
    if (!dbOptions || typeof dbOptions !== 'object') return [];
    return Object.entries(dbOptions)
      .sort(([a], [b]) => a.localeCompare(b))
      .map(([key, value]) => ({
        key,
        text: typeof value === 'object' && value.text ? value.text : String(value)
      }));
  }

  // Transform full question
  transformQuestion(q) {
    return {
      id: q.id,
      question_text: q.question_text,
      options: this.transformOptions(q.options),
      correct_answer: q.correct_answer,
      explanation: q.explanation || "No explanation available.",
      difficulty: q.difficulty || "Medium",
      question_image_url: q.question_image_url,
      marks: q.marks || 1,
      is_important: q.is_important || false
    };
  }

  // Fetch MCQs - handles all cases
  async fetchMCQs({ topicId, chapterId, chapterOnly = false }) {
    let query = this.supabase
      .from("questions")
      .select("*")
      .or("question_format.eq.single_choice,question_format.eq.multiple_choice,question_format.is.null")
      .not("options", "is", null)
      .order("created_at");

    if (topicId) {
      // TOPIC LEVEL
      query = query.eq("topic_id", topicId);
    } else if (chapterId && chapterOnly) {
      // CHAPTER LEVEL ONLY (topic_id is NULL)
      query = query.eq("chapter_id", chapterId).is("topic_id", null);
    } else if (chapterId) {
      // ALL questions in chapter
      query = query.eq("chapter_id", chapterId);
    }

    const { data, error } = await query;
    if (error) throw error;
    
    return (data || []).map(q => this.transformQuestion(q));
  }

  // Get question counts by difficulty
  async fetchQuestionCounts({ topicId, chapterId, chapterOnly = false }) {
    let query = this.supabase
      .from("questions")
      .select("difficulty")
      .or("question_format.eq.single_choice,question_format.eq.multiple_choice,question_format.is.null")
      .not("options", "is", null);

    if (topicId) {
      query = query.eq("topic_id", topicId);
    } else if (chapterId && chapterOnly) {
      query = query.eq("chapter_id", chapterId).is("topic_id", null);
    } else if (chapterId) {
      query = query.eq("chapter_id", chapterId);
    }

    const { data, error } = await query;
    if (error) throw error;

    const counts = { all: 0, Low: 0, Medium: 0, Intermediate: 0, Advanced: 0 };
    (data || []).forEach(q => {
      counts.all++;
      if (q.difficulty && counts[q.difficulty] !== undefined) {
        counts[q.difficulty]++;
      }
    });

    return counts;
  }
}

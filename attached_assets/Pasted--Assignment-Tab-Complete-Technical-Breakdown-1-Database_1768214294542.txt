📚 Assignment Tab - Complete Technical Breakdown
1. Database Schema
assignments Table
Column	Type	Description
id	uuid	Primary key
subject_id	uuid	Subject reference
chapter_id	uuid	Chapter reference
topic_id	uuid	Topic reference (NULL = chapter-level)
title	text	Assignment title
description	text	Description
instructions	text	Student instructions
questions	jsonb	Array of questions (stored as JSON)
total_marks	integer	Total marks
passing_marks	integer	Minimum passing marks
duration_minutes	integer	Time limit
due_date	timestamp	Deadline
valid_until	timestamptz	Expiry date
is_active	boolean	Visibility
source_type	text	"ai_generated" or "manual"
ai_generation_config	jsonb	AI config if auto-generated
assignment_submissions Table
Column	Type	Description
id	uuid	Primary key
assignment_id	uuid	Assignment reference
student_id	uuid	Student user ID
answers	jsonb	Student answers with grading
score	integer	Total marks obtained
percentage	numeric	Score percentage
feedback	text	Overall feedback
submitted_at	timestamp	Submission time
graded_at	timestamp	Grading time
time_taken_seconds	integer	Time spent

Grading Rules:

MCQ/True-False: Full marks or 0 (exact match)
Short/Long Answer: Partial marks allowed (meaning-based)
Empty answers: 0 marks
Fallback: If AI fails, marked as "Pending manual review"

class AssignmentService {
  // Fetch assignments for a subject/chapter/topic
  async fetchAssignments({ subjectId, chapterId, topicId }) {
    let query = supabase
      .from('assignments')
      .select('id, title, description, due_date, total_marks, passing_marks, duration_minutes, chapter_id, topic_id')
      .eq('subject_id', subjectId)
      .eq('is_active', true)
      .order('created_at', { ascending: false });
    
    if (chapterId) query = query.eq('chapter_id', chapterId);
    if (topicId) query = query.eq('topic_id', topicId);
    
    return await query;
  }

  // Fetch assignment details with questions
  async fetchAssignmentDetails(assignmentId) {
    const { data } = await supabase
      .from('assignments')
      .select('*')
      .eq('id', assignmentId)
      .single();
    
    return {
      ...data,
      questions: data.questions.map((q, idx) => ({
        id: q.id || `q_${idx}`,
        question: q.question,
        type: q.type,
        options: q.options || [],
        marks: q.marks,
        image_url: q.image_url,
      }))
    };
  }

  // Submit assignment
  async submitAssignment({ assignmentId, answers, score, percentage, timeTaken }) {
    const user = await supabase.auth.getUser();
    
    return await supabase
      .from('assignment_submissions')
      .upsert({
        assignment_id: assignmentId,
        student_id: user.data.user.id,
        answers: answers,
        score: score,
        percentage: percentage,
        submitted_at: new Date().toISOString(),
        graded_at: new Date().toISOString(),
        time_taken_seconds: timeTaken,
      }, { onConflict: 'assignment_id,student_id' });
  }

  // Grade subjective answers with AI
  async gradeWithAI(questions) {
    const { data, error } = await supabase.functions.invoke('ai-grade-assignment', {
      body: { questions }
    });
    return data;
  }

  // Extract answer from image
  async extractFromImage(imageUrl, questionContext) {
    const { data, error } = await supabase.functions.invoke('extract-answer-from-image', {
      body: { image_url: imageUrl, question_context: questionContext }
    });
    return data;
  }
}

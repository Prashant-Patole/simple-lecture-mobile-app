DPP (Daily Practice Problems) Tab - Complete Documentation
Overview
The DPP tab allows students to take 10 random practice questions on a topic/chapter daily. It tracks streak, history, and shows progress on a calendar.

Database Schema
1. dpp_questions - Stores all DPP questions
Column	Type	Description
id	uuid	Primary key
document_id	uuid	Reference to DPP document
subject_id	uuid	Subject this question belongs to
chapter_id	uuid	Chapter (can be NULL for topic-level)
topic_id	uuid	Topic (NULL for chapter-level questions)
dpp_number	integer	DPP set number
question_number	integer	Question order in set
question_text	text	Question text (supports LaTeX)
options	jsonb	Options like {"a": "text", "b": "text", ...}
correct_answer	text	Correct option (e.g., "A", "B")
explanation	text	Explanation for the answer
difficulty	text	"easy", "medium", "hard"
is_active	boolean	Whether question is active
created_at	timestamp	Creation time

2. dpp_topic_submissions - Stores student submissions

total_questions	integer	Total questions (usually 10)

3. dpp_attempted_questions - Tracks which questions user has seen
Column	Type	Description

Topic Level vs Chapter Level DPP
Aspect	Topic Level	Chapter Level
Filter	topic_id = ?	chapter_id = ? AND topic_id IS NULL
Use Case	Questions specific to one topic	Questions covering full chapter
Priority	Checked first	Fallback if topic has < 5 questions
Query Logic:


// 1. Try topic-level first
if (topicId) {
  const questions = await fetchQuestions('topic_id', topicId);
  if (questions.length >= 5) return questions;
}

// 2. Fallback to chapter-level
if (chapterId) {
  const questions = await fetchQuestions('chapter_id', chapterId);
  if (questions.length >= 5) return questions;
}
Streak Calculation Algorithm

const calculateStreak = (submissions) => {
  const today = format(new Date(), 'yyyy-MM-dd');
  const yesterday = format(subDays(new Date(), 1), 'yyyy-MM-dd');
  
  // Get unique dates, sorted newest first
  const dates = [...new Set(submissions.map(s => s.test_date))].sort().reverse();
  
  // Must have submission today or yesterday to have streak
  if (dates[0] !== today && dates[0] !== yesterday) return 0;
  
  let streak = 1;
  for (let i = 0; i < dates.length - 1; i++) {
    const current = parseISO(dates[i]);
    const next = parseISO(dates[i + 1]);
    const diffDays = differenceInDays(current, next);
    
    if (diffDays === 1) streak++;  // Consecutive days
    else break;  // Gap found
  }
  return streak;
};

Mobile API Integration
For mobile integration, you would use direct Supabase queries:


// DPP Service for Mobile
class DPPService {
  constructor(supabaseClient) {
    this.supabase = supabaseClient;
  }

  // Check if DPP available for topic
  async checkDPPAvailability(topicId, chapterId) {
    // Try topic level
    if (topicId) {
      const { count } = await this.supabase
        .from('dpp_questions')
        .select('*', { count: 'exact', head: true })
        .eq('topic_id', topicId)
        .eq('is_active', true)
        .neq('correct_answer', '')
        .not('correct_answer', 'is', null);
      
      if (count > 0) return { available: true, source: 'topic', count };
    }
    
    // Try chapter level
    if (chapterId) {
      const { count } = await this.supabase
        .from('dpp_questions')
        .select('*', { count: 'exact', head: true })
        .eq('chapter_id', chapterId)
        .eq('is_active', true)
        .neq('correct_answer', '')
        .not('correct_answer', 'is', null);
      
      if (count > 0) return { available: true, source: 'chapter', count };
    }
    
    return { available: false, source: null, count: 0 };
  }

  // Fetch 10 random DPP questions
  async fetchDPPQuestions(userId, topicId, chapterId) {
    // Get attempted question IDs
    const { data: attempted } = await this.supabase
      .from('dpp_attempted_questions')
      .select('question_id')
      .eq('student_id', userId)
      .eq('topic_id', topicId);
    
    const attemptedIds = (attempted || []).map(a => a.question_id);

    // Build query
    let query = this.supabase
      .from('dpp_questions')
      .select('*')
      .eq('is_active', true)
      .eq('topic_id', topicId)
      .neq('correct_answer', '')
      .not('correct_answer', 'is', null);
    
    if (attemptedIds.length > 0) {
      query = query.not('id', 'in', `(${attemptedIds.join(',')})`);
    }
    
    const { data: questions } = await query.limit(50);
    
    // Shuffle and select 10
    return this.shuffleAndSelect(questions || [], 10);
  }

  // Submit DPP
  async submitDPP(userId, topicId, questions, answers, score, timeSeconds) {
    const today = new Date().toISOString().split('T')[0];
    
    const { data, error } = await this.supabase
      .from('dpp_topic_submissions')
      .upsert({
        student_id: userId,
        topic_id: topicId,
        test_date: today,
        dpp_type: 'teacher',
        questions: questions,
        answers: answers,
        score: score,
        total_questions: questions.length,
        time_taken_seconds: timeSeconds,
        submitted_at: new Date().toISOString()
      }, { onConflict: 'student_id,topic_id,test_date' });
    
    return { data, error };
  }

  // Get submission history
  async getSubmissionHistory(userId, topicId) {
    const { data } = await this.supabase
      .from('dpp_topic_submissions')
      .select('test_date, score, total_questions, time_taken_seconds')
      .eq('student_id', userId)
      .eq('topic_id', topicId)
      .order('test_date', { ascending: false });
    
    return data || [];
  }

  shuffleAndSelect(array, count) {
    const shuffled = [...array].sort(() => Math.random() - 0.5);
    return shuffled.slice(0, count);
  }
}

1. Subject Selection Flow
When you select a subject in the Learning page:

Step	Component	Action
1	Learning.tsx	User selects a subject from sidebar
2	Learning.tsx	Passes subjectName prop to AITeachingAssistant
3	AITeachingAssistant.tsx	Receives subjectName, topicId, chapterId
4	TeacherAvatarPanel.tsx	Fetches appropriate avatar based on language
5	Edge Function	Uses subjectName to validate question relevance

// Avatar fetching based on language (gender selection)
useEffect(() => {
  const fetchAvatar = async () => {
    const gender = language === 'hi-IN' ? 'female' : 'male';
    
    const { data } = await supabase
      .from('counselor_avatars')
      .select('*')
      .eq('gender', gender)
      .eq('is_active', true)
      .order('display_order')
      .limit(1)
      .maybeSingle();

    if (data) {
      setAvatarUrl(data.image_url);
      setAvatarName(data.name);
    }
  };
  fetchAvatar();

Step 1: Subject Relevance Check

// Check if question is relevant to the subject
function isQuestionRelevant(question: string, subjectName: string): boolean {
  const subjectKeywords: Record<string, string[]> = {
    'physics': ['force', 'motion', 'energy', 'velocity', 'acceleration', ...],
    'chemistry': ['atom', 'molecule', 'element', 'compound', 'reaction', ...],
    'mathematics': ['equation', 'formula', 'algebra', 'geometry', ...],
    'biology': ['cell', 'organism', 'dna', 'gene', 'protein', ...],
  };
  
  // Returns true if question contains subject-related keywords
  return hasRelevantKeyword || mentionsSubject || isEducational;
}

6. Response Structure

interface TeachingResponse {
  cached: boolean;
  answer: string;
  presentationSlides: PresentationSlide[];  // 6 slides
  latexFormulas: LatexFormula[];
  keyPoints?: string[];
  followUpQuestions?: string[];
  narrationText: string;
  subjectName?: string;
  detectedTopic?: string;
  relatedConcepts?: string[];
}

interface PresentationSlide {
  title: string;
  content: string;
  keyPoints: string[];
  formula?: string;          // LaTeX formula
  narration: string;         // 150-250 words for TTS
  isStory?: boolean;         // Slide 5: Story slide
  isTips?: boolean;          // Slide 6: Tips slide
  infographic?: string;      // Description for image gen
  infographicUrl?: string;   // Generated image URL
  videoUrl?: string;         // Optional video for story
}

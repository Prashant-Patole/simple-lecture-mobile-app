// React Native Example
import RazorpayCheckout from 'react-native-razorpay';

const handlePayment = async () => {
  // Step 1: Create order
  const orderResponse = await fetch(
    'https://oxwhqvsoelqqsblmqkxx.supabase.co/functions/v1/create-topup-order',
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${accessToken}`,
      },
      body: JSON.stringify({
        userId,
        courseId,
        languages: selectedLanguages,
      }),
    }
  );
  
  const { razorpayOrderId, razorpayKeyId, orderId, amount } = await orderResponse.json();

  // Step 2: Open Razorpay checkout
  const options = {
    key: razorpayKeyId,
    amount: amount * 100,  // Convert to paise
    currency: 'INR',
    order_id: razorpayOrderId,
    name: 'SimpleLecture',
    description: `${selectedLanguages.length} Language Top-Up`,
    prefill: { email: userEmail },
    theme: { color: '#8B5CF6' },
  };

  try {
    const paymentData = await RazorpayCheckout.open(options);
    
    // Step 3: Verify payment
    await fetch(
      'https://oxwhqvsoelqqsblmqkxx.supabase.co/functions/v1/verify-topup-payment',
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${accessToken}`,
        },
        body: JSON.stringify({
          razorpay_order_id: paymentData.razorpay_order_id,
          razorpay_payment_id: paymentData.razorpay_payment_id,
          razorpay_signature: paymentData.razorpay_signature,
          orderId,
          userId,
          courseId,
          languages: selectedLanguages,
        }),
      }
    );
    
    // Success - navigate back
  } catch (error) {
    // Handle error or cancellation
  }
};

Building the Hierarchy Tree:


function buildCategoryHierarchy(categories) {
  const categoryMap = new Map();
  const level1Categories = [];

  // First pass: create all category objects
  categories.forEach((cat) => {
    categoryMap.set(cat.id, {
      id: cat.id,
      name: cat.name,
      slug: cat.slug,
      icon: cat.icon,
      level: cat.level,
      display_order: cat.display_order || 0,
      subcategories: [],
    });
  });

  // Second pass: build hierarchy
  categories.forEach((cat) => {
    const category = categoryMap.get(cat.id);
    if (!category) return;

    if (cat.level === 1) {
      // Level 1: Add to root array
      level1Categories.push(category);
    } else if (cat.parent_id) {
      // Level 2 or 3: Add to parent's subcategories
      const parent = categoryMap.get(cat.parent_id);
      if (parent) {
        parent.subcategories.push(category);
      }
    }
  });

  // Sort subcategories by display_order
  level1Categories.forEach((cat) => {
    cat.subcategories.sort((a, b) => a.display_order - b.display_order);
    cat.subcategories.forEach((subcat) => {
      subcat.subcategories.sort((a, b) => a.display_order - b.display_order);
    });
  });

  return level1Categories.sort((a, b) => a.display_order - b.display_order);
}

4. Direct Supabase Query - Courses by Category Hierarchy
For cascading category filtering (courses from parent include all children):


// Get courses for a specific category AND all its descendants
async function getCoursesByHierarchy(parentCategoryId, subCategoryId, subSubCategoryId) {
  
  // Case 1: Sub-sub category selected (Level 3) - direct filter
  if (subSubCategoryId) {
    const { data } = await supabase
      .from('course_categories')
      .select(`
        courses (
          id, name, slug, thumbnail_url, short_description,
          price_inr, original_price_inr, duration_months,
          student_count, rating, is_active
        )
      `)
      .eq('category_id', subSubCategoryId);
    
    return data?.map(cc => cc.courses).filter(c => c?.is_active);
  }
  
  // Case 2: Sub-category selected (Level 2) - include Level 3 children
  if (subCategoryId) {
    // Get sub-sub categories (Level 3)
    const { data: level3Cats } = await supabase
      .from('categories')
      .select('id')
      .eq('parent_id', subCategoryId)
      .eq('is_active', true);
    
    const categoryIds = [subCategoryId, ...(level3Cats?.map(c => c.id) || [])];
    
    const { data } = await supabase
      .from('course_categories')
      .select('courses(*)')
      .in('category_id', categoryIds);
    
    // Deduplicate courses
    const uniqueCourses = new Map();
    data?.forEach(cc => {
      if (cc.courses?.is_active) {
        uniqueCourses.set(cc.courses.id, cc.courses);
      }
    });
    
    return Array.from(uniqueCourses.values());
  }
  
  // Case 3: Parent category selected (Level 1) - include Level 2 & 3
  if (parentCategoryId) {
    // Get all descendants
    const { data: level2Cats } = await supabase
      .from('categories')
      .select('id')
      .eq('parent_id', parentCategoryId)
      .eq('is_active', true);
    
    const level2Ids = level2Cats?.map(c => c.id) || [];
    
    const { data: level3Cats } = await supabase
      .from('categories')
      .select('id')
      .in('parent_id', level2Ids)
      .eq('is_active', true);
    
    const allCategoryIds = [
      parentCategoryId,
      ...level2Ids,
      ...(level3Cats?.map(c => c.id) || [])
    ];
    
    const { data } = await supabase
      .from('course_categories')
      .select('courses(*)')
      .in('category_id', allCategoryIds);
    
    // Deduplicate
    const uniqueCourses = new Map();
    data?.forEach(cc => {
      if (cc.courses?.is_active) {
        uniqueCourses.set(cc.courses.id, cc.courses);
      }
    });
    
    return Array.from(uniqueCourses.values());
  }
  
  // Case 4: No category - return popular courses
  const { data } = await supabase
    .from('courses')
    .select('*')
    .eq('is_active', true)
    .order('student_count', { ascending: false })
    .limit(50);
  
  return data;
}

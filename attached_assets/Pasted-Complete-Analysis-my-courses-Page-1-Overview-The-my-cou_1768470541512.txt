Complete Analysis: /my-courses Page
1. Overview
The /my-courses page displays all courses a student is enrolled in with:

Category-based filtering
Progress tracking
Course thumbnails and details
Navigation to learning page
2. Database Schema Used
enrollments Table
Column	Type	Description
id	uuid	Primary key
student_id	uuid	Reference to auth user
course_id	uuid	Reference to courses
enrolled_at	timestamp	Enrollment date
expires_at	timestamp	Subscription expiry
is_active	boolean	Active enrollment flag
batch_id	uuid	Optional batch reference
courses Table
Column	Type	Description
id	uuid	Primary key
name	text	Course name
slug	text	URL-friendly identifier
thumbnail_url	text	Course image
short_description	text	Brief description
duration_months	integer	Course duration
price_inr	integer	Course price
course_categories Table (Junction Table)
Column	Type	Description
id	uuid	Primary key
course_id	uuid	Reference to courses
category_id	uuid	Reference to categories
categories Table
Column	Type	Description
id	uuid	Primary key
name	text	Category name
slug	text	URL identifier
icon	text	Emoji/icon
parent_id	uuid	Parent category reference
level	integer	Hierarchy level (1=parent)
display_order	integer	Sort order
is_active	boolean	Active status
chapters Table
Column	Type	Description
id	uuid	Primary key
course_id	uuid	Reference to courses
subject	text	Subject name
title	text	Chapter title
student_progress Table
Column	Type	Description
id	uuid	Primary key
student_id	uuid	Reference to auth user
chapter_id	uuid	Reference to chapters
is_completed	boolean	Completion status
score	integer	Progress score
3. API Endpoints Used
Get User
Supabase Auth → supabase.auth.getUser()
Fetch Enrollments with Course Details

GET /rest/v1/enrollments?
  select=id,enrolled_at,course_id,courses(id,name,slug,thumbnail_url,short_description,duration_months,price_inr)
  &student_id=eq.{userId}
  &is_active=eq.true
Fetch Parent Categories (for filter tabs)

GET /rest/v1/categories?
  select=id,name,slug,icon
  &is_active=eq.true
  &level=eq.1
  &order=display_order
Fetch Course Categories

GET /rest/v1/course_categories?
  select=course_id,category_id,categories(id,name,slug,parent_id,level)
  &course_id=in.({courseIds})
Fetch Chapters

GET /rest/v1/chapters?
  select=id,course_id
  &course_id=in.({courseIds})
Fetch Parent Categories

GET /rest/v1/categories?
  select=id,name,slug
  &id=in.({parentIds})
Fetch Student Progress

GET /rest/v1/student_progress?
  select=chapter_id,is_completed
  &student_id=eq.{userId}
  &chapter_id=in.({chapterIds})

8. Progress Calculation Logic

// From useEnrolledCoursesWithCategories.ts

// 1. Get all chapters for the course
const courseChapterIds = chaptersByCourse[course.id] || [];

// 2. Calculate progress
let progress = 0;
if (courseChapterIds.length > 0) {
  // Count completed chapters from student_progress
  const completed = courseChapterIds.filter(id => 
    completedChapters.has(id)
  ).length;
  
  // Calculate percentage
  progress = Math.round((completed / courseChapterIds.length) * 100);
}
9. Category Filtering Implementation

// State for selected category
const [selectedCategory, setSelectedCategory] = useState<string>('all');

// Get unique parent categories from enrolled courses
const enrolledCategories = useMemo(() => {
  if (!enrolledCourses) return [];
  const categoryIds = new Set(
    enrolledCourses.map(c => c.parentCategoryId).filter(Boolean)
  );
  return categories?.filter(cat => categoryIds.has(cat.id)) || [];
}, [enrolledCourses, categories]);

// Filter courses by selected category
const filteredCourses = useMemo(() => {
  if (!enrolledCourses) return [];
  if (selectedCategory === 'all') return enrolledCourses;
  return enrolledCourses.filter(c => c.parentCategoryId === selectedCategory);
}, [enrolledCourses, selectedCategory]);
